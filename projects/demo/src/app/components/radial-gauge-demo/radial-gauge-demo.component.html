<lib-radial-gauge
  [value]="{{ value() }}"
  [min]="{{ min() }}"
  [max]="{{ max() }}"
  [size]="{{ size() }}"
  [outerThickness]="{{ outerThickness() }}"
  [innerThickness]="{{ innerThickness() }}"
  [gap]="{{ gap() }}"
  [segmentGapPx]="{{ segmentGapPx() }}"
  [segments]="customSegments"
  title="My Gauge"
  description="Units"
/>

  to: number;
  color: string;
{{ '}' }}

];
